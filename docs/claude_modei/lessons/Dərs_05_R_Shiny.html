5.1. Köməkçi funksiyalar — db_helpers.R

cat > R/db_helpers.R << 'EOF'
# ═══════════════════════════════════════════════════════════════
# R/db_helpers.R — Verilənlər bazası ilə iş funksiyaları
# ═══════════════════════════════════════════════════════════════

# Layihə xülasəsini al
get_project_summary <- function(pool) {
  dbGetQuery(pool, "SELECT * FROM reporting.v_project_summary")
}

# Regional aktivlik
get_regional_activity <- function(pool) {
  data <- dbGetQuery(pool, "SELECT * FROM reporting.v_regional_activity")
  # Koordinatlar əlavə et (joinlə)
  data %>% 
    left_join(REGION_COORDS, by = c("region_name" = "code"))
}

# Büdcə sapması
get_budget_variance <- function(pool, project_id = NULL) {
  sql <- "SELECT * FROM reporting.v_budget_variance"
  if (!is.null(project_id)) {
    sql <- paste0(sql, " WHERE project_id = $1")
    return(dbGetQuery(pool, sql, params = list(project_id)))
  }
  dbGetQuery(pool, sql)
}

# Podratçı performansı
get_contractor_performance <- function(pool) {
  dbGetQuery(pool, "SELECT * FROM reporting.v_contractor_performance")
}

# Aylıq progress
get_monthly_progress <- function(pool) {
  dbGetQuery(pool, "SELECT * FROM reporting.v_monthly_progress")
}

# KPI aqreqatları
get_kpi_stats <- function(pool) {
  dbGetQuery(pool, "
    SELECT
      COUNT(*) AS total_projects,
      COUNT(*) FILTER (WHERE status = 'in_progress') AS active_projects,
      COUNT(*) FILTER (WHERE status = 'completed') AS completed,
      COUNT(*) FILTER (WHERE deadline_status = 'overdue') AS overdue,
      COALESCE(SUM(total_budget_planned), 0) AS total_budget,
      COALESCE(AVG(progress_percent), 0)::int AS avg_progress
    FROM reporting.v_project_summary
  ")
}

8 Modul 1: KPI kartları

İlk tab — rəhbərliyin "bir baxışda" görmək istədiyi mühüm rəqəmlər.

shiny/modules/mod_kpi.R
cat > modules/mod_kpi.R << 'EOF'
# ═══════════════════════════════════════════════════════════════
# mod_kpi.R — Ümumi baxış KPI kartları
# ═══════════════════════════════════════════════════════════════

mod_kpi_ui <- function(id) {
  ns <- NS(id)
  tagList(
    fluidRow(
      bs4ValueBoxOutput(ns("total"), width = 3),
      bs4ValueBoxOutput(ns("active"), width = 3),
      bs4ValueBoxOutput(ns("budget"), width = 3),
      bs4ValueBoxOutput(ns("overdue"), width = 3)
    ),
    fluidRow(
      bs4Card(
        title = "Status üzrə paylanma", width = 6, maximizable = TRUE,
        plotlyOutput(ns("status_pie"), height = "350px")
      ),
      bs4Card(
        title = "Obyekt növü üzrə", width = 6, maximizable = TRUE,
        plotlyOutput(ns("type_bar"), height = "350px")
      )
    ),
    fluidRow(
      bs4Card(
        title = "Son layihələr", width = 12,
        DTOutput(ns("projects_table"))
      )
    )
  )
}

mod_kpi_server <- function(id, pool) {
  moduleServer(id, function(input, output, session) {

    # ═══ Reactive data ═══
    kpi <- reactive({
      get_kpi_stats(pool)
    })

    projects <- reactive({
      get_project_summary(pool)
    })

    # ═══ Value boxes ═══
    output$total <- renderbs4ValueBox({
      bs4ValueBox(
        value = kpi()$total_projects,
        subtitle = "Ümumi layihə",
        icon = icon("folder-tree"),
        color = "primary",
        footer = "Bütün statuslar"
      )
    })

    output$active <- renderbs4ValueBox({
      bs4ValueBox(
        value = kpi()$active_projects,
        subtitle = "Aktiv layihə",
        icon = icon("person-digging"),
        color = "success",
        footer = "in_progress statusunda"
      )
    })

    output$budget <- renderbs4ValueBox({
      bs4ValueBox(
        value = paste0(format(round(kpi()$total_budget / 1e6, 1)), "M AZN"),
        subtitle = "Planlı büdcə",
        icon = icon("sack-dollar"),
        color = "warning",
        footer = paste("Orta progress:", kpi()$avg_progress, "%")
      )
    })

    output$overdue <- renderbs4ValueBox({
      bs4ValueBox(
        value = kpi()$overdue,
        subtitle = "Gecikən",
        icon = icon("triangle-exclamation"),
        color = if (kpi()$overdue > 0) "danger" else "success",
        footer = "son bitmə tarixini keçənlər"
      )
    })

    # ═══ Status pie ═══
    output$status_pie <- renderPlotly({
      d <- projects() %>%
        count(status) %>%
        mutate(label = recode(status,
          "in_progress" = "İcrada",
          "completed" = "Tamamlanıb",
          "planned" = "Planlaşdırılıb",
          "cancelled" = "Ləğv",
          "suspended" = "Dayandırılıb"
        ))

      plot_ly(d, labels = ~label, values = ~n, type = "pie",
              marker = list(colors = c("#f59e0b","#10b981","#64748b","#ef4444","#d97706"))) %>%
        layout(showlegend = TRUE, margin = list(l = 20, r = 20, t = 10, b = 10))
    })

    # ═══ Obyekt tipi bar ═══
    output$type_bar <- renderPlotly({
      d <- projects() %>%
        count(object_type) %>%
        arrange(desc(n))

      plot_ly(d, x = ~object_type, y = ~n, type = "bar",
              marker = list(color = "#7c3aed")) %>%
        layout(xaxis = list(title = ""), yaxis = list(title = "Layihə sayı"))
    })

    # ═══ Cədvəl ═══
    output$projects_table <- renderDT({
      d <- projects() %>%
        select(project_code, project_name, region, status, progress_percent, total_budget_planned)

      datatable(d,
        options = list(pageLength = 10, scrollX = TRUE),
        rownames = FALSE,
        colnames = c("Kod","Layihə","Region","Status","Progress %","Büdcə")
      ) %>%
        formatCurrency("total_budget_planned", currency = " AZN", before = FALSE, digits = 0) %>%
        formatStyle("progress_percent",
          background = styleColorBar(c(0,100), "#c4b5fd"),
          backgroundSize = "98% 88%",
          backgroundRepeat = "no-repeat",
          backgroundPosition = "center")
    })
  })
}

# R/claude_api.R — Claude API wrapper

ask_claude <- function(system_prompt, user_prompt, max_tokens = 2048) {
  api_key <- Sys.getenv("ANTHROPIC_API_KEY")
  model <- Sys.getenv("ANTHROPIC_MODEL", "claude-sonnet-4-6")

  if (api_key == "") {
    stop("ANTHROPIC_API_KEY .Renviron-da təyin edilməyib")
  }

  req <- httr2::request("https://api.anthropic.com/v1/messages") %>%
    httr2::req_headers(
      "x-api-key" = api_key,
      "anthropic-version" = "2023-06-01",
      "content-type" = "application/json"
    ) %>%
    httr2::req_body_json(list(
      model = model,
      max_tokens = max_tokens,
      system = system_prompt,
      messages = list(list(role = "user", content = user_prompt))
    )) %>%
    httr2::req_timeout(60)

  resp <- httr2::req_perform(req)

  if (httr2::resp_status(resp) != 200) {
    stop("Claude API xətası: ", httr2::resp_status(resp))
  }

  body <- httr2::resp_body_json(resp)
  body$content[[1]]$text
}

  • Yazma əməliyyatları — backend (auth + validate + trigger-lər işləsin)
  • AI təhlil — backend (API açarı orada, cache orada)
  • İstifadəçi idarəsi — backend (JWT və bcrypt orada)
