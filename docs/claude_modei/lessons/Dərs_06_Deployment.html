1UFW firewall konfiqurasiyası
bash — firewall qaydaları
# Default deny (hər şey bağlı)
sudo ufw default deny incoming
sudo ufw default allow outgoing

# Yalnız SSH və HTTPS-ə icazə
sudo ufw allow ssh
sudo ufw allow 80/tcp    # HTTP (Let's Encrypt üçün lazımdır)
sudo ufw allow 443/tcp   # HTTPS

# Tailscale interface-i üçün icazə (tailscale0)
sudo ufw allow in on tailscale0

# Aktivləşdir

Niyə 3001, 5173, 3838 portlarını bağlayırıq?

Node, Vite, Shiny portlarına birbaşa müraciəti bağlayırıq — bütün trafik Nginx-dən (80/443) keçməlidir. Bu həm təhlükəsizlik, həm də SSL üçün lazımdır. Daxili networkdə (localhost) bu portlar işləyəcək, amma kənardan əlçatmaz olacaq.

2fail2ban ilə SSH brute-force qorunması

[DEFAULT]
bantime = 3600
findtime = 600
maxretry = 5

[sshd]
enabled = true
port = ssh
filter = sshd
logpath = /var/log/auth.log
EOF

sudo systemctl restart fail2ban
sudo systemctl enable fail2ban

# Status
sudo fail2ban-client status sshd

9 Nginx reverse proxy

İndi ən vacib hissəyə gəlirik — Nginx-in 3 servisi (Node, static React, Shiny) bir domendə birləşdirməsi.

1Nginx quraşdırma
bash
sudo apt install -y nginx
sudo systemctl enable nginx
sudo systemctl status nginx
2Site konfiqurasiyası
/etc/nginx/sites-available/ttis
sudo tee /etc/nginx/sites-available/ttis > /dev/null << 'EOF'
# ═══════════════════════════════════════════════════════════════
# TTİS — Reverse proxy config
# ═══════════════════════════════════════════════════════════════

# Upstream-lər
upstream ttis_backend {
    server 127.0.0.1:3001;
    keepalive 32;
}

upstream ttis_shiny {
    server 127.0.0.1:3838;
}

# HTTP → HTTPS redirect (SSL-dən sonra aktiv olacaq)
server {
    listen 80;
    server_name ttis.arti.edu.az;

    # Let's Encrypt challenge üçün
    location /.well-known/acme-challenge/ {
        root /var/www/html;
    }

    # Qalan hər şey HTTPS-ə yönləndir
    location / {
        return 301 https://$host$request_uri;
    }
}

# HTTPS əsas server
server {
    listen 443 ssl http2;
    server_name ttis.arti.edu.az;

    # SSL sertifikatları (Let's Encrypt qoyacaq)
    # ssl_certificate     /etc/letsencrypt/live/ttis.arti.edu.az/fullchain.pem;
    # ssl_certificate_key /etc/letsencrypt/live/ttis.arti.edu.az/privkey.pem;

    # Təhlükəsizlik header-ləri
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;
    add_header Strict-Transport-Security "max-age=31536000" always;

    # Gzip sıxılma
    gzip on;
    gzip_vary on;
    gzip_types text/plain text/css application/json application/javascript
               text/xml application/xml application/xml+rss text/javascript
               image/svg+xml;

    # Max upload (fayl yükləmə üçün)
    client_max_body_size 50M;

    # ═══ 1) React frontend (static) ═══
    root /var/www/html/ttis;
    index index.html;

    location / {
        try_files $uri $uri/ /index.html;

        # Asset-lər üçün aggressive cache (hash-lı fayllar)
        location ~* \.(js|css|woff2|png|jpg|svg)$ {
            expires 1y;
            add_header Cache-Control "public, immutable";
        }
    }

    # ═══ 2) Backend API ═══
    location /api/ {
        proxy_pass http://ttis_backend;
        proxy_http_version 1.1;
        proxy_set_header Connection "";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_read_timeout 60s;
    }

    # ═══ 3) R Shiny ═══
    location /shiny/ {
        rewrite ^/shiny/(.*)$ /ttis/$1 break;

        proxy_pass http://ttis_shiny;
        proxy_redirect off;

        # WebSocket (Shiny real-time üçün vacib!)
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_read_timeout 20d;

        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }

    # Nginx loglar
    access_log /var/log/nginx/ttis.access.log;
    error_log  /var/log/nginx/ttis.error.log;
}
EOF

# Aktiv et
sudo ln -sf /etc/nginx/sites-available/ttis /etc/nginx/sites-enabled/ttis

# Default siteni sil
sudo rm -f /etc/nginx/sites-enabled/default

# Sintaksis yoxla və restart
sudo nginx -t
sudo systemctl reload nginx

12 Avtomatik backup

Production-da ən dəyərli şey — məlumatdır. Gündəlik backup və 30 günlük retention.

bash — /usr/local/bin/ttis-backup.sh
sudo tee /usr/local/bin/ttis-backup.sh > /dev/null << 'EOF'
#!/bin/bash
# TTİS — Gündəlik avtomatik backup

set -euo pipefail

BACKUP_DIR=/var/backups/ttis
TIMESTAMP=$(date +%Y%m%d_%H%M%S)
DB_DUMP="$BACKUP_DIR/db_${TIMESTAMP}.dump"

mkdir -p "$BACKUP_DIR"

# ═══ 1. PostgreSQL dump ═══
echo "[$(date)] 🔄 DB backup başladı"
PGPASSWORD="$DB_PASSWORD" pg_dump \
  -h localhost -U ttis_app -d ttis_db -Fc \
  -f "$DB_DUMP"

# ═══ 2. Config faylları ═══
tar czf "$BACKUP_DIR/configs_${TIMESTAMP}.tar.gz" \
  /etc/nginx/sites-available/ttis \
  /var/www/ttis/backend/.env \
  /var/www/ttis/frontend/.env.production \
  2>/dev/null || true

# ═══ 3. 30 gündən köhnə fayllar silinsin ═══
find "$BACKUP_DIR" -type f -mtime +30 -delete

# ═══ 4. Yoxlama ═══
SIZE=$(du -h "$DB_DUMP" | cut -f1)
echo "[$(date)] ✅ Backup uğurlu: $DB_DUMP ($SIZE)"
EOF

# DB şifrəsi üçün env fayl
sudo tee /etc/default/ttis-backup > /dev/null << 'EOF'
DB_PASSWORD="YOUR_STRONG_DB_PASSWORD"
EOF
sudo chmod 600 /etc/default/ttis-backup

sudo chmod +x /usr/local/bin/ttis-backup.sh

# Test işə sal
sudo -E bash -c 'source /etc/default/ttis-backup && /usr/local/bin/ttis-backup.sh'

14 GitHub Actions CI/CD

Hər dəfə git push etdikdə tətbiqin avtomatik deploy olunması.

.github/workflows/deploy.yml
# Mac-da layihə kökündə yarat
mkdir -p .github/workflows

cat > .github/workflows/deploy.yml << 'EOF'
name: Deploy to tt-server

on:
  push:
    branches: [main]

jobs:
  deploy:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '22'

      # ═══ Backend testlər ═══
      - name: Backend install and test
        working-directory: ./backend
        run: |
          npm ci
          # npm test  # əgər testləriniz varsa

      # ═══ Frontend build ═══
      - name: Frontend install and build
        working-directory: ./frontend
        env:
          VITE_API_URL: https://ttis.arti.edu.az/api/v1
        run: |
          npm ci
          npm run build

      # ═══ Tailscale ilə VM-ə qoşulmaq ═══
      - name: Connect to Tailscale
        uses: tailscale/github-action@v2
        with:
          oauth-client-id: ${{ secrets.TS_OAUTH_CLIENT_ID }}
          oauth-secret: ${{ secrets.TS_OAUTH_SECRET }}
          tags: tag:ci

      # ═══ VM-ə deploy ═══
      - name: Deploy to VM
        env:
          SSH_KEY: ${{ secrets.SSH_PRIVATE_KEY }}
        run: |
          echo "$SSH_KEY" > /tmp/deploy_key
          chmod 600 /tmp/deploy_key

          # Backend kodunu köçür
          rsync -avz --delete \
            -e "ssh -i /tmp/deploy_key -o StrictHostKeyChecking=no" \
            --exclude node_modules --exclude .env \
            backend/ tariyel@tt-server:/var/www/ttis/backend/

          # Frontend build
          rsync -avz --delete \
            -e "ssh -i /tmp/deploy_key -o StrictHostKeyChecking=no" \
            frontend/dist/ tariyel@tt-server:/var/www/html/ttis/

          # Shiny
          rsync -avz --delete \
            -e "ssh -i /tmp/deploy_key -o StrictHostKeyChecking=no" \
            --exclude .Renviron \
            shiny/ tariyel@tt-server:/srv/shiny-server/ttis/

          # Backend yenilə
          ssh -i /tmp/deploy_key -o StrictHostKeyChecking=no \
            tariyel@tt-server \
            'cd /var/www/ttis/backend && npm ci --omit=dev && pm2 reload ttis-backend'

      - name: Notify
        if: success()
        run: echo "✅ Deploy uğurludur"
EOF
